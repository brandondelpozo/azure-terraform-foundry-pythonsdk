# SAS URL Upload Test using curl commands

# Step 1: Generate SAS URL
curl -X POST "https://func-testfoundry3-endpoint.azurewebsites.net/api/generate-upload-url" \
  -H "Content-Type: application/json" \
  -d '{"filename": "working_test.docx"}'

# Step 2: Upload file (replace SAS_URL from step 1 response)
curl -X PUT "SAS_URL_FROM_STEP_1" \
  -H "x-ms-blob-type: BlockBlob" \
  -H "Content-Type: application/vnd.openxmlformats-officedocument.wordprocessingml.document" \
  --data-binary @working_test.docx
